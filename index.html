2- Definir una función que reciba tres parámetros como argumentos 
y devuelva el valor mayor de entre ellos.	@author: cyn """

def max_de_tres(a,b,c):
	aux = 0
	numeros = [a,b,c]
		for num in numeros:
			if num > aux :
				aux = num
		print(f"El numero Mayor es {aux}")
		return

max_de_tres(3,7,2)
